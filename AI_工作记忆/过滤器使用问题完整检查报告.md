### 推荐的过滤器使用模式

```python
# 根据翻译类型选择合适的过滤器
def get_appropriate_filter(translation_type: TranslationType) -> callable:
    filter_rules = AdvancedFilterRules()
    
    if translation_type == TranslationType.KEYED:
        return lambda text, key="": filter_rules.should_translate_keyed(text, key)
    elif translation_type == TranslationType.DEFINJECTED:
        return lambda text, field, def_type="": filter_rules.should_translate_def_field(text, field, def_type)
    else:
        return lambda text, context="": filter_rules.should_translate_corpus(text, context)
```
